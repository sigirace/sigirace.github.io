```python
from abc import ABC, abstractmethod

class Workable(ABC):
    @abstractmethod
    def work(self):
        pass


class Eatable(ABC):
    @abstractmethod
    def eat(self):
        pass


class Sleepable(ABC):
    @abstractmethod
    def sleep(self):
        pass


class Human(Workable, Eatable, Sleepable):
    def work(self):
        print("Human Working...")

    def eat(self):
        print("Human Eating...")

    def sleep(self):
        print("Human Sleeping...")


class Robot(Workable):
    def work(self):
        print("Robot Working...")
```
